create table account(username varchar(15) not null, password varchar(32) not null, email varchar(255) not null, primary key (username) ) charset=utf8;


create table daily (username varchar(15) not null, date timestamp not null, content text not null, title varchar(255) not null, subject varchar(255) not null, foreign key(username) references account(username) ) charset=utf8;


create table account_role (username varchar(15) not null, role varchar(15) not null, primary key (username, role) ) charset=utf8;

create table java6_class (id int auto_increment primary key, name char(100) not null, packageName char(100) not null, src char(100) not null) charset=utf8;

create table java6_package (id int auto_increment primary key, name char(100) not null, src char(100) not null) charset=utf8;

create table user_comment (id int auto_increment primary key, name char(15) not null, date timestamp not null, context text not null) charset=utf8;
