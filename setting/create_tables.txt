create table account(name varchar(15) not null, password varchar(255) not null, email varchar(255) not null, primary key (name) ) charset=utf8;


create table daily (name varchar(15) not null, date timestamp not null, content text not null, title varchar(255) not null, subject varchar(255) not null, foreign key (name) references t_account(name) ) charset=utf8;

create table account_role (name varchar(15) not null, role varchar(15) not null, primary key (name, role) ) charset=utf8;

create table java6_class (id int auto_increment primary key, name char(100) not null, packageName char(100) not null, src char(100) not null) charset=utf8;

create table java6_package (id int auto_increment primary key, name char(100) not null, src char(100) not null) charset=utf8;

create table user_comment (id int auto_increment primary key, username char(15) not null, date timestamp not null, content text not null) charset=utf8;

create table user_bookmark (id int auto_increment primary key, username char(15) not null, date timestamp not null, url text not null, title text not null, imgUrl text not null) charset=utf8;

create table account_log(id int auto_increment primary key, username char(15) not null, message char(255) not null, date timestamp not null) charset=utf8;
